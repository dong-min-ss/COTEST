#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>
#include <queue>
#include <stack>
using namespace std;
typedef long long ll;

const int dy[] = { -1,0,1,0 };
const int dx[] = { 0,1,0,-1 };

ll n, a, b;
vector<pair<ll, ll>> v;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL); cout.tie(NULL);

    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> a >> b;
        v.push_back({ a,b });
    }
    v.push_back(v[0]);

    ll res1 = 0, res2 = 0;
    for (int i = 0; i < n; i++) { // 신발끈 정리
        res1 += v[i].first * v[i + 1].second;
        res2 += v[i].second * v[i + 1].first;
    }

    cout << fixed;
    cout.precision(1);
    cout<< abs(res1 - res2) / 2.0;

    return 0;
}
