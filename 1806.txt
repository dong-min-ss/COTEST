#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>
#include <queue>
#include <map>
using namespace std;
typedef long long ll;

const int dy[] = { 0, -1,0,1,0 };
const int dx[] = { 0, 0,1,0,-1 };

int n, m, a[100001];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL); cout.tie(NULL);

    cin >> n >> m;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    int _min = n + 1;
    int start = 0, sum = 0;

    for (int end = 0; end < n; end++) {
        sum += a[end];

        while (sum >= m) {
            _min = min(_min, end - start + 1);
            sum -= a[start];
            start++;
        }
    }
    cout << (_min == n + 1 ? 0 : _min);

    return 0;
}
