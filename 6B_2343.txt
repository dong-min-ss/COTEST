#include<iostream>
#include<vector>
#include<deque>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;

int a[100001], psum[100001], n, m, ret = 987654321;

bool check(int mid) {
    int cnt = 0, start = 0;
    while (true) {
        for (int i = start; i <= n; i++) {
            if (psum[i] - psum[start] > mid) {
                start = i - 1;
                cnt++;
                break;
            }
            if (i == n) {
                cnt++;
                return cnt <= m;
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    int left = 1, right = 0;
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        psum[i] = psum[i - 1] + a[i];
        right += a[i];
        left = max(left, a[i]);
    }

    while (left <= right) {
        int mid = (left + right) / 2;
        if (check(mid)) {
            ret = min(mid, ret);
            right = mid - 1;
        }
        else left = mid + 1;
    }
    cout << ret;

    return 0;
}
