#include<iostream>
#include<vector>
#include<map>
#include<cstring>
#include<algorithm>
#include<climits>
#include <iomanip>
using namespace std;

int n, LIS[1001], len, num, cnt[1001];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);


    cin >> n;
    vector<int> v(n);

    for(int i = 0; i < n; i++) {
        cin >> v[i];
    }

    for (int i = 0; i < n; i++) {
        int _max = 0;
        for (int j = 0; j < i; j++) {
            if (v[i] > v[j] && cnt[j] > _max)
                _max = cnt[j];
        }
        cnt[i] = _max + 1;
        len = max(len, cnt[i]);
    }
    cout << len;
    
    return 0;
}
